 * è§£æ Markdown æµ‹è¯•ç”¨ä¾‹æ–‡æ¡£
 * @param content Markdown æ–‡æœ¬å†…å®¹
 * @returns è§£æç»“æœ
 */
export function parseMarkdownTestCases(content: string): ParseResult {
  const lines = content.split('\n');
  const categories: string[][] = [];
  const testCases: ParsedTestCase[] = [];

  let currentCategory: string[] = [];
  let currentTestCase: Partial<ParsedTestCase> | null = null;
  let currentField: string | null = null;

  for (let i = 0; i < lines.length; i++) {
    const line = lines[i];

    // åŒ¹é…äºŒçº§æ ‡é¢˜ï¼ˆå¤§åˆ†ç±»ï¼‰
    if (line.startsWith('## ')) {
      const categoryName = line.replace('## ', '').replace(/^\d+\.\s*/, '').trim();
      if (categoryName && !categoryName.includes('æµ‹è¯•æ¦‚è¿°')) {
        currentCategory = [categoryName];
      }
      continue;
    }

    // åŒ¹é…ä¸‰çº§æ ‡é¢˜ï¼ˆå­åˆ†ç±»ï¼‰
    if (line.startsWith('### ')) {
      const subCategoryName = line.replace('### ', '').replace(/^\d+\.\d+\s*/, '').trim();
      if (subCategoryName && currentCategory.length > 0) {
        currentCategory = [currentCategory[0], subCategoryName];
        if (!categories.some(c => c.join('/') === currentCategory.join('/'))) {
          categories.push([...currentCategory]);
        }
      }
      continue;
    }

    // åŒ¹é…å››çº§æ ‡é¢˜ï¼ˆæµ‹è¯•ç”¨ä¾‹ï¼‰
    const testCaseMatch = line.match(/^####\s+(TC-[A-Z]+-\d+)[:ï¼š]\s*(.+)$/);
    if (testCaseMatch) {
      // ä¿å­˜ä¸Šä¸€ä¸ªç”¨ä¾‹
      if (currentTestCase && currentTestCase.code) {
        testCases.push(currentTestCase as ParsedTestCase);
      }

      currentTestCase = {
        code: testCaseMatch[1],
        title: testCaseMatch[2].trim(),
        categoryPath: [...currentCategory],
        priority: 'P1',
        steps: [],
        expectedResults: [],
      };
      currentField = null;
      continue;
    }

    // è§£æç”¨ä¾‹å­—æ®µ
    if (currentTestCase) {
      // ä¼˜å…ˆçº§
      if (line.includes('**ä¼˜å…ˆçº§**')) {
        const priorityMatch = line.match(/P[012]/);
        if (priorityMatch) {
          currentTestCase.priority = priorityMatch[0] as Priority;
        }
        continue;
      }

      // å‰ç½®æ¡ä»¶
      if (line.includes('**å‰ç½®æ¡ä»¶**')) {
        const value = line.split(':').slice(1).join(':').trim();
        currentTestCase.preconditions = value;
        currentField = 'preconditions';
        continue;
      }

      // æµ‹è¯•æ­¥éª¤
      if (line.includes('**æµ‹è¯•æ­¥éª¤**')) {
        currentField = 'steps';
        continue;
      }

      // é¢„æœŸç»“æœ
      if (line.includes('**é¢„æœŸç»“æœ**')) {
        currentField = 'expectedResults';
        continue;
      }

      // æµ‹è¯•æ•°æ®
      if (line.includes('**æµ‹è¯•æ•°æ®**')) {
        const value = line.split(':').slice(1).join(':').trim();
        currentTestCase.testData = value;
        currentField = 'testData';
        continue;
      }

      // è§£æåˆ—è¡¨é¡¹
      const listItemMatch = line.match(/^\s*[\d\-\*]+[.\)]\s*(.+)$/);
      if (listItemMatch && currentField) {
        const value = listItemMatch[1].trim();
        if (currentField === 'steps') {
          currentTestCase.steps?.push(value);
        } else if (currentField === 'expectedResults') {
          currentTestCase.expectedResults?.push(value);
        }
      }
    }
  }

  // ä¿å­˜æœ€åä¸€ä¸ªç”¨ä¾‹
  if (currentTestCase && currentTestCase.code) {
    testCases.push(currentTestCase as ParsedTestCase);
  }

  return { categories, testCases };
}
